
ancestry {
	id int pk
	ancestry varchar
}

ontology {
	id varchar pk
	name varchar
	uri varchar
}

traits {
	trait_name varchar
	study int > studies.id
	ontology varchar > ontology.name
	sample_size int
	ncase int
	ncontrol int
	units varchar
	ancestry int > ancestry.id
	type varchar
	id int pk
}

variants {
	id int pk
	chr int
	pos int
	a1 varchar
	a2 varchar
	sas_freq float
	eas_freq float
	eur_freq float
	afr_freq float
	his_freq float
	sas_ldscore float
	eas_ldscore float
	eur_ldscore float
	afr_ldscore float
	his_ldscore float
}

ld {
	id int pk
	id1 int > variants.id
	id2 int > variants.id
	ancestry int > ancestry.id
	r_eur float
	r_sas float
	r_eas float
	r_afr float
	r_his float
}

finemap {
	id int pk
	leadsnp int > variants.id
	trait int > traits.id
}

assocs {
	id int pk
	variant int > variants.id
	trait int > finemap.id
	beta float
	se float
	pval float
}

regions {
	id int pk
	name varchar
	chr int
	start int
	end int
	ancestry int > ancestry.id
}

coloc {
	id int pk
	region int > regions.id
	iteration int
	trait int > finemap.id
	candidate_snp int > variants.id
	posterior_prob float
	regional_prob float
	posterior_explained_by_snp float
}
